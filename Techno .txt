document.getElementById('cropSelect').addEventListener('change', function() {
  const crop = this.value;
  const varietySelect = document.getElementById('varietySelect');
  varietySelect.innerHTML = '<option value="">-- اختر الصنف --</option>';
  if(varieties[crop]) {
    Object.keys(varieties[crop]).forEach(v => {
      const option = document.createElement('option');
      option.value = v;
      option.textContent = v;
      varietySelect.appendChild(option);
    });
  }
  document.getElementById('featuresDisplay').innerHTML = '';
});

document.getElementById('varietySelect').addEventListener('change', function() {
  const crop = document.getElementById('cropSelect').value;
  const variety = this.value;
  if(varieties[crop] && varieties[crop][variety]) {
    document.getElementById('featuresDisplay').innerHTML = varieties[crop][variety];
  } else {
    document.getElementById('featuresDisplay').innerHTML = '';
  }
});
